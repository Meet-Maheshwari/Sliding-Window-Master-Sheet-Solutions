class Solution {
public:
    int kDistinctChar(string& s, int k) {
        //your code goes here
        unordered_map<char, int> map;

        int l = 0, r = 0, maxLen = 0;

        while(r < s.size()) {
            map[s[r]]++;

            if(map.size() > k) {
                map[s[l]]--;
                if(map[s[l]] == 0) { map.erase(s[l]); }
                l++;
            }

            if(map.size() <= k) {
                int len = r - l + 1;
                maxLen = max(len, maxLen);
            }
            r++;
        } 
        return maxLen;
    }
};
